  const response = await fetch('/analyze', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ code, language })
  });
  
  if (!response.ok) {
    const error = await response.json();
    console.error('Error:', error.error);
    return null;
  }
  
  return await response.json();
}
